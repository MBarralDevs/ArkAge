# Plan A — Contracts + Indexer + Schema Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deploy ArkAge's 5 smart contracts to Arc Testnet, stand up the Postgres schema in Neon, wire managed event indexing (Goldsky Mirror + Circle Contract Platform webhooks) to stream chain events into Postgres, and add the workflow-stuck reconciliation cron — producing the foundational data layer that Plans B/C/D build on.

**Architecture:**
- Smart contracts in Foundry at `contracts/`. Five immutable contracts (HookComposer, ReputationHook, PolicyHook, EvaluatorFeeHook, AgentRegistry) deployed via deterministic CREATE2.
- Next.js 16 App Router app at repo root. Hosts the webhook receiver routes and the reconciliation cron. Dashboard + MCP server come in later plans.
- Prisma + Neon Postgres. Event-sourced + materialized state schema per spec §7.
- Goldsky Mirror streams canonical contract events (USDC, ERC-8183, ERC-8004 ×3). Circle Contract Platform webhooks push our 5 contracts' events to a dedicated Vercel route.

**Tech Stack:**
- Foundry (forge + cast + anvil) for smart contracts
- Next.js 16 (App Router) + Node.js 24 LTS on Vercel
- Prisma 6 + Neon Postgres
- viem 2 for RPC
- Vitest for TS tests
- @circle-fin/smart-contract-platform SDK for webhook setup
- Goldsky Mirror for canonical-contract indexing
- Vercel Cron for reconciliation

**Plan reference:** Spec at `docs/superpowers/specs/2026-05-02-arkage-design.md` §1 (architecture), §2 (contracts), §7 (schema), §4.7 (indexing infra), §11 (pre-impl checklist).

**File responsibilities (key files only):**

- `contracts/src/AgentRegistry.sol` — operator+policy+per-job-fee registry, owner-gated by ERC-8004 NFT ownership
- `contracts/src/HookComposer.sol` — pure router, no state beyond ordered hook arrays, ACP-only caller check
- `contracts/src/PolicyHook.sol` — `beforeAction` gate; reads from AgentRegistry; reverts on violations
- `contracts/src/ReputationHook.sol` — `afterAction` writer to ERC-8004 ReputationRegistry
- `contracts/src/EvaluatorFeeHook.sol` — `afterAction` USDC pull from provider to treasury
- `src/lib/db.ts` — Prisma client singleton with Vercel/Edge-safe pattern
- `src/lib/chain.ts` — viem public + wallet clients for Arc Testnet
- `src/lib/circle-webhook-verify.ts` — verify HMAC signature on incoming webhook
- `src/app/api/webhooks/circle/route.ts` — POST handler, verifies signature, dispatches to ingest worker
- `src/workers/ingest-circle-event.ts` — parses event, writes to appropriate Postgres table, fires `resumeHook` for matching workflow tokens
- `src/workers/reconcile-stuck-workflows.ts` — scans `workflow_runs` for stale `last_advanced_at`, queries chain, fires synthesized resumeHook
- `prisma/schema.prisma` — full §7 schema as Prisma models

// =============================================================
// Identity & wallets
// =============================================================

model Builder {
  id             BigInt   @id @default(autoincrement())
  primaryWallet  Bytes    @unique @map("primary_wallet")
  displayName    String?  @map("display_name")
  createdAt      DateTime @default(now()) @map("created_at")
  updatedAt      DateTime @updatedAt @map("updated_at")
  wallets        Wallet[]

  @@map("builders")
}

model Wallet {
  id              BigInt   @id @default(autoincrement())
  address         Bytes    @unique
  tier            Int      @db.SmallInt
  custody         String   // 'modular' | 'dcw' | 'system'
  accountType     String   @map("account_type") // 'msca' | 'eoa'
  builderId       BigInt?  @map("builder_id")
  circleWalletId  String?  @map("circle_wallet_id")
  status          String   @default("active") // 'active' | 'paused' | 'revoked'
  createdAt       DateTime @default(now()) @map("created_at")
  updatedAt       DateTime @updatedAt @map("updated_at")
  builder         Builder? @relation(fields: [builderId], references: [id])
  agentsAsOperator Agent[]

  @@map("wallets")
}

model Agent {
  id                       BigInt           @id @default(autoincrement())
  agentId                  Decimal          @unique @map("agent_id") @db.Decimal(78, 0)

  currentMetadataId        BigInt?          @map("current_metadata_id")
  currentPolicyId          BigInt?          @map("current_policy_id")
  agentWalletAddress       Bytes            @map("agent_wallet_address")
  registeredAtBlock        BigInt           @map("registered_at_block")
  active                   Boolean          @default(true)
  createdAt                DateTime         @default(now()) @map("created_at")
  updatedAt                DateTime         @updatedAt @map("updated_at")
  currentOperatorWallet    Wallet           @relation(fields: [currentOperatorWalletId], references: [id])
  metadata                 AgentMetadata[]
  policies                 Policy[]

  jobsAsProvider           Job[]            @relation("ProviderAgent")

  reputationValidations    ReputationValidation[]
  x402Endpoints            X402Endpoint[]
  x402SessionsAsBuyer      X402Session[]    @relation("BuyerAgent")
  x402SessionsAsSeller     X402Session[]    @relation("SellerAgent")

model Job {
  id                BigInt   @id @default(autoincrement())
  jobId             Decimal  @unique @map("job_id") @db.Decimal(78, 0)
  clientAgentId     BigInt   @map("client_agent_id")
  providerAgentId   BigInt?  @map("provider_agent_id")
  evaluatorAddress  Bytes    @map("evaluator_address")
  evaluatorTier     String?  @map("evaluator_tier")
  status            String   // 'open' | 'funded' | 'submitted' | 'completed' | 'rejected' | 'expired'
  budget            Decimal? @db.Decimal(38, 0) // USDC ERC-20, 6 decimals
  evaluatorFee      Decimal? @map("evaluator_fee") @db.Decimal(38, 0) // USDC ERC-20, 6 decimals
  descriptionUri    String?  @map("description_uri")
  descriptionHash   Bytes?   @map("description_hash")
  hookAddress       Bytes    @map("hook_address")
  expiredAt         DateTime @map("expired_at")
  reasonHash        Bytes?   @map("reason_hash")
  createdAtBlock    BigInt?  @map("created_at_block")
  completedAtBlock  BigInt?  @map("completed_at_block")
  createdAt         DateTime @default(now()) @map("created_at")
  updatedAt         DateTime @updatedAt @map("updated_at")
  clientAgent       Agent    @relation("ClientAgent", fields: [clientAgentId], references: [id])
  providerAgent     Agent?   @relation("ProviderAgent", fields: [providerAgentId], references: [id])
  events            JobEvent[]
  evaluations       JobEvaluation[]

  @@index([clientAgentId, status, createdAt(sort: Desc)])
  @@index([providerAgentId, status, createdAt(sort: Desc)])
  @@map("jobs")
}

model JobEvent {
  id          BigInt   @id @default(autoincrement())
  jobId       BigInt   @map("job_id")
  eventKind   String   @map("event_kind")

  blockTime   DateTime @map("block_time")
  createdAt   DateTime @default(now()) @map("created_at")

  @@map("job_events")

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import {IAgentRegistry} from "./interfaces/IAgentRegistry.sol";
import {IIdentityRegistry} from "./interfaces/IIdentityRegistry.sol";
import {IACP} from "./interfaces/IACP.sol";

contract AgentRegistry is IAgentRegistry {
    address public immutable IDENTITY_REGISTRY;
    address public immutable AGENTIC_COMMERCE;

    mapping(uint256 => AgentInfo) private _agents;
    mapping(address => uint256) public override agentIdByOperator;

    mapping(uint256 => uint256) public override evaluatorFeeFor;
    mapping(uint256 => bool) public override jobFeeRecorded;

    event AgentRegistered(uint256 indexed agentId, address indexed operator, bytes32 policyHash);
    event OperatorUpdated(uint256 indexed agentId, address indexed previous, address indexed next);
    event PolicyUpdated(uint256 indexed agentId, bytes32 policyHash, uint128 perTxCap, uint64 evalFeeMax);
    event AgentDeactivated(uint256 indexed agentId);
    event AgentReactivated(uint256 indexed agentId);
    event JobFeeRecorded(uint256 indexed jobId, uint256 fee);

    modifier onlyIdentityOwner(uint256 agentId) {
        require(IIdentityRegistry(IDENTITY_REGISTRY).ownerOf(agentId) == msg.sender, "not identity owner");
        _;
    }

    constructor(address identityRegistry, address agenticCommerce) {
        IDENTITY_REGISTRY = identityRegistry;
        AGENTIC_COMMERCE = agenticCommerce;
    }

    function agents(uint256 agentId) external view override returns (AgentInfo memory) {
        return _agents[agentId];
    }

    function agentByOperator(address op) external view override returns (AgentInfo memory) {
        uint256 agentId = agentIdByOperator[op];
        return _agents[agentId];
    }

    function registerAgent(uint256 agentId, address op, bytes32 policy, uint128 perTx, uint64 evalFeeMax)
        external
        override
        onlyIdentityOwner(agentId)
    {
        require(_agents[agentId].operatorWallet == address(0), "already registered");
        require(op != address(0), "operator zero");
        _agents[agentId] = AgentInfo({
            operatorWallet: op,
            currentPolicyHash: policy,
            perTxCap: perTx,
            evaluatorFeeMax: evalFeeMax,
            active: true
        });
        agentIdByOperator[op] = agentId;
        emit AgentRegistered(agentId, op, policy);
    }

    function updateOperator(uint256 agentId, address op) external override onlyIdentityOwner(agentId) {
        require(op != address(0), "operator zero");
        address prev = _agents[agentId].operatorWallet;
        require(prev != address(0), "not registered");
        agentIdByOperator[prev] = 0;
        _agents[agentId].operatorWallet = op;
        agentIdByOperator[op] = agentId;
        emit OperatorUpdated(agentId, prev, op);
    }

    function updatePolicy(uint256 agentId, bytes32 policy, uint128 perTx, uint64 evalFeeMax)
        external
        override
        onlyIdentityOwner(agentId)
    {
        require(_agents[agentId].operatorWallet != address(0), "not registered");
        _agents[agentId].currentPolicyHash = policy;
        _agents[agentId].perTxCap = perTx;
        _agents[agentId].evaluatorFeeMax = evalFeeMax;
        emit PolicyUpdated(agentId, policy, perTx, evalFeeMax);
    }

    function deactivate(uint256 agentId) external override onlyIdentityOwner(agentId) {
        _agents[agentId].active = false;
        emit AgentDeactivated(agentId);
    }

    function reactivate(uint256 agentId) external override onlyIdentityOwner(agentId) {
        require(_agents[agentId].operatorWallet != address(0), "not registered");
        _agents[agentId].active = true;
        emit AgentReactivated(agentId);
    }

    function recordJobFee(uint256 jobId, uint256 fee) external override {
        require(!jobFeeRecorded[jobId], "fee already recorded");

        IACP.Job memory j = IACP(AGENTIC_COMMERCE).getJob(jobId);
        require(j.client == msg.sender, "not job client");

        uint256 clientAgentId = agentIdByOperator[msg.sender];
        require(clientAgentId != 0, "client not registered");
        require(fee <= _agents[clientAgentId].evaluatorFeeMax, "fee exceeds max");

        evaluatorFeeFor[jobId] = fee;
        jobFeeRecorded[jobId] = true;
        emit JobFeeRecorded(jobId, fee);
    }
}

import {IACPHook} from "./interfaces/IACPHook.sol";

contract PolicyHook is IACPHook {
    address public immutable AGENTIC_COMMERCE;
    address public immutable AGENT_REGISTRY;

    constructor(address acp, address agentRegistry) {
        AGENTIC_COMMERCE = acp;
        AGENT_REGISTRY = agentRegistry;
    }

    function beforeAction(uint256 jobId, bytes4 selector, bytes calldata data) external override {
        require(msg.sender == AGENTIC_COMMERCE, "only ACP");

        IACP.Job memory job = IACP(AGENTIC_COMMERCE).getJob(jobId);
        address actor = _resolveActor(selector, job);

        IAgentRegistry.AgentInfo memory info = IAgentRegistry(AGENT_REGISTRY).agentByOperator(actor);

        // If actor isn't a registered ArkAge agent, skip policy gate (BYO flows allowed).
        if (info.operatorWallet == address(0)) return;

        require(info.active, "policy: agent inactive");

        if (selector == IACP.fund.selector) {
            uint256 fundAmount = abi.decode(data, (uint256));
            require(fundAmount <= info.perTxCap, "policy: per-tx cap");
        }
        // Additional selector-specific checks added as needed.
    }

    function afterAction(uint256, bytes4, bytes calldata) external pure override {
        // no-op
    }

    function _resolveActor(bytes4 selector, IACP.Job memory job) internal pure returns (address) {
        if (selector == IACP.fund.selector) return job.client;
        if (selector == IACP.setBudget.selector) return job.provider;
        if (selector == IACP.submit.selector) return job.provider;
        if (selector == IACP.complete.selector) return job.evaluator;
        if (selector == IACP.reject.selector) return job.evaluator;
        if (selector == IACP.setProvider.selector) return job.client;

import {IACPHook} from "./interfaces/IACPHook.sol";
import {IACP} from "./interfaces/IACP.sol";
import {IReputationRegistry} from "./interfaces/IReputationRegistry.sol";
import {IAgentRegistry} from "./interfaces/IAgentRegistry.sol";

contract ReputationHook is IACPHook {
    address public immutable AGENTIC_COMMERCE;
    address public immutable REPUTATION_REGISTRY;
    address public immutable AGENT_REGISTRY;

    string private constant TAG_SRC = "src:acp";
    string private constant TAG_COMPLETE = "outcome:complete";
    string private constant TAG_REJECT = "outcome:reject";

    constructor(address acp, address reputationRegistry, address agentRegistry) {
        AGENTIC_COMMERCE = acp;
        REPUTATION_REGISTRY = reputationRegistry;
        AGENT_REGISTRY = agentRegistry;
    }

    function beforeAction(uint256, bytes4, bytes calldata) external pure override {
        // no-op
    }

    function afterAction(uint256 jobId, bytes4 selector, bytes calldata) external override {
        require(msg.sender == AGENTIC_COMMERCE, "only ACP");

        IACP.Job memory job = IACP(AGENTIC_COMMERCE).getJob(jobId);
        uint256 providerAgentId = IAgentRegistry(AGENT_REGISTRY).agentIdByOperator(job.provider);
        if (providerAgentId == 0) return; // unknown provider — silent skip

        if (selector == IACP.complete.selector) {
            IReputationRegistry(REPUTATION_REGISTRY).giveFeedback(
                providerAgentId,
                int128(100),
                uint8(0),
                TAG_SRC,
                TAG_COMPLETE,
                _jobEndpoint(jobId),
                _evidenceURI(job.reason),
                job.reason
            );
        } else if (selector == IACP.reject.selector) {
            IReputationRegistry(REPUTATION_REGISTRY).giveFeedback(
                providerAgentId,
                int128(-50),
                uint8(0),
                TAG_SRC,
                TAG_REJECT,
                _jobEndpoint(jobId),
                _evidenceURI(job.reason),
                job.reason
            );
        }
    }

    function _jobEndpoint(uint256 jobId) internal pure returns (string memory) {
        // Placeholder URI scheme; ArkAge dashboard parses this format.
        return string(abi.encodePacked("arkage://job/", _toString(jobId)));
    }

    function _evidenceURI(bytes32 reason) internal pure returns (string memory) {
        return string(abi.encodePacked("arkage://evidence/", _toHexString(reason)));
    }

    function _toString(uint256 n) internal pure returns (string memory) {
        if (n == 0) return "0";
        uint256 j = n;
        uint256 len;
        while (j != 0) { len++; j /= 10; }
        bytes memory b = new bytes(len);
        while (n != 0) { len--; b[len] = bytes1(uint8(48 + (n % 10))); n /= 10; }
        return string(b);
    }

    function _toHexString(bytes32 v) internal pure returns (string memory) {
        bytes memory chars = "0123456789abcdef";
        bytes memory s = new bytes(64);
        for (uint256 i = 0; i < 32; i++) {
            s[i * 2] = chars[uint8(v[i] >> 4)];
            s[i * 2 + 1] = chars[uint8(v[i] & 0x0f)];
        }
        return string(s);
    }
}

import { createHmac, timingSafeEqual } from "node:crypto";

export function verifyCircleWebhook(
  rawBody: string,
  receivedSignatureHex: string,
  secret: string
): boolean {
  const expected = createHmac("sha256", secret).update(rawBody).digest();

  let received: Buffer;
  try {
    received = Buffer.from(receivedSignatureHex, "hex");
  } catch {
    return false;
  }

  if (received.length !== expected.length) return false;
  return timingSafeEqual(expected, received);
}

import { NextResponse } from "next/server";
import { verifyCircleWebhook } from "@/lib/circle-webhook-verify";
import { ingestCircleEvent, type CircleWebhookPayload } from "@/workers/ingest-circle-event";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

export async function POST(request: Request): Promise<Response> {
  const secret = process.env.CIRCLE_WEBHOOK_SECRET;
  if (!secret) {
    return NextResponse.json({ error: "webhook secret not configured" }, { status: 500 });
  }

  const signature = request.headers.get("x-circle-signature");
  if (!signature) {
    return NextResponse.json({ error: "missing signature" }, { status: 401 });
  }

  const rawBody = await request.text();
  if (!verifyCircleWebhook(rawBody, signature, secret)) {
    return NextResponse.json({ error: "invalid signature" }, { status: 401 });
  }

  let payload: CircleWebhookPayload;
  try {
    payload = JSON.parse(rawBody);
  } catch {
    return NextResponse.json({ error: "invalid json" }, { status: 400 });
  }

  await ingestCircleEvent(payload);
  return NextResponse.json({ ok: true });
}

import { db } from "@/lib/db";

export interface StuckWorkflow {
  runId: string;
  kind: string;
  kindId: bigint;
  lastAdvancedAt: Date;
}

export async function findStuckWorkflows(opts: {
  olderThanMinutes: number;
}): Promise<StuckWorkflow[]> {
  const threshold = new Date(Date.now() - opts.olderThanMinutes * 60 * 1000);
  const rows = await db.workflowRun.findMany({
    where: {
      status: "running",
      lastAdvancedAt: { lt: threshold },
    },
    select: { runId: true, kind: true, kindId: true, lastAdvancedAt: true },
    take: 100,
  });
  return rows;
}

export async function reconcileStuckWorkflows(): Promise<{ scanned: number; advanced: number }> {
  const stuck = await findStuckWorkflows({ olderThanMinutes: 10 });

  let advanced = 0;
  for (const run of stuck) {
    // Plan B will: query chain state via viem, fire synthesized resumeHook
    // Plan A just records that we noticed.
    await db.auditLog.create({
      data: {
        actorKind: "system",
        actorId: "stuck-workflow-reconciler",
        action: "stuck_detected",
        targetKind: "workflow_run",
        targetId: run.runId,
        payloadJsonb: { kind: run.kind, kindId: String(run.kindId) },
      },
    });
    advanced++;
  }

  return { scanned: stuck.length, advanced };
}

  chainHeadBlock: bigint;

export async function reconcileIndexerCursor(): Promise<CursorReport[]> {

  const head = await publicClient.getBlockNumber();

  const reports: CursorReport[] = [];
  for (const c of cursors) {
    const lag = head - BigInt(c.lastBlock.toString());
    reports.push({
      source: c.source,
      contractAddress: "0x" + Buffer.from(c.contractAddress).toString("hex"),
      lastIndexedBlock: BigInt(c.lastBlock.toString()),
      chainHeadBlock: head,
      lagBlocks: lag,
    });

    if (lag > 100n) {
      await db.auditLog.create({
        data: {
          actorKind: "system",
          actorId: "indexer-cursor-reconciler",
          action: "indexer_lag_warning",
          targetKind: "indexer_cursor",
          targetId: `${c.source}:${c.contractAddress.toString("hex")}`,
          payloadJsonb: { lagBlocks: lag.toString() } as object,
        },
      });
    }
  }

  return reports;
}
